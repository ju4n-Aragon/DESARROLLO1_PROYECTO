-- Script SQL para crear la base de datos en PostgreSQL
-- Este script asume que estás conectado a PostgreSQL como superusuario o con permisos para crear bases de datos.
-- Puedes ejecutarlo en psql o en una herramienta como pgAdmin.

-- Crear la base de datos (si no existe)
CREATE DATABASE consultores_db
    WITH 
    OWNER = postgres  -- Cambia 'postgres' por tu usuario si es necesario
    ENCODING = 'UTF8'
    LC_COLLATE = 'en_US.UTF-8'
    LC_CTYPE = 'en_US.UTF-8'
    TABLESPACE = pg_default
    CONNECTION LIMIT = -1;

-- Conéctate a la nueva base de datos (en psql: \c consultores_db)
-- Ahora, crea el esquema de tablas

-- Tabla principal de usuarios
-- Representa la clase base 'Usuario' del diagrama, con herencia para roles (Cliente, Consultor, Admin)
CREATE TABLE IF NOT EXISTS usuarios (
    id SERIAL PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,  -- En producción, usa hashing (e.g., con pg_crypto), aquí es plain para simplicidad
    nombre VARCHAR(100) NOT NULL,
    rol VARCHAR(20) NOT NULL CHECK (rol IN ('admin', 'cliente', 'consultor'))
);

-- Tabla para detalles específicos de Consultores (herencia de Usuario)
-- Atributos como tarifa y especialidad del diagrama
CREATE TABLE IF NOT EXISTS consultores (
    id_usuario INTEGER PRIMARY KEY REFERENCES usuarios(id) ON DELETE CASCADE,
    tarifa DECIMAL(10, 2) NOT NULL CHECK (tarifa > 0),
    especialidad VARCHAR(50) NOT NULL
);

-- Tabla para reservas (citas)
-- Representa la clase 'Cita' del diagrama, con relaciones a Cliente y Consultor
CREATE TABLE IF NOT EXISTS reservas (
    id SERIAL PRIMARY KEY,
    id_cliente INTEGER NOT NULL REFERENCES usuarios(id) ON DELETE RESTRICT ON UPDATE CASCADE,
    id_consultor INTEGER NOT NULL REFERENCES usuarios(id) ON DELETE RESTRICT ON UPDATE CASCADE,
    fecha TIMESTAMP NOT NULL CHECK (fecha > CURRENT_TIMESTAMP),  -- No permite fechas pasadas, como en el código
    estado VARCHAR(20) NOT NULL DEFAULT 'Activa' CHECK (estado IN ('Activa', 'Cancelada', 'Completada'))  -- Enum implícito del diagrama, agregué 'Completada' para escalabilidad
);

-- Índices para optimización (opcional pero recomendado)
CREATE INDEX idx_usuarios_username ON usuarios(username);
CREATE INDEX idx_reservas_fecha ON reservas(fecha);
CREATE INDEX idx_reservas_cliente ON reservas(id_cliente);
CREATE INDEX idx_reservas_consultor ON reservas(id_consultor);
